Go 并发编程篇（4）-任务编排 WaitGroup

发表于2023-12-15|更新于2025-03-18|golang

前言

鸟窝的《深入理解 Go 并发编程》读书笔记以及其他 go 并发编程知识学习的汇总，持续完善中。


sync.WaitGroup

允许一个 goroutine 等待其他多个 goroutine 执行完毕后再继续执行。

用法

func main() {
  var (
    wg   sync.WaitGroup
    urls = []string{`baidu.com`, `bing.com`}
  )
  
  wg.add(len(urls))
  for i := range urls {
    go func(url string) {
      defer wg.Done()
      
      resp, err := http.Get(url)
      if err != nil {
        return
      }
    }(url)
  }
  wg.Wait()
}

实现

sync.WaitGroup 通过维护一个等待的 goroutine 数量，利用原子操作保证并发安全，使用信号量机制实现 goroutine 的阻塞和唤醒，从而实现了等待一组 goroutine 完成的功能。

type WaitGroup struct {
  noCopy noCopy       // 用于辅助 vet 工具检查是否可以通过 copy 复制 WaitGroup
  // 为了保证并发安全，WaitGroup 在更新等待的 goroutine 数量时使用了原子操作。
  // 原子操作可以确保在多 goroutine 环境下，对共享变量的读写操作是线程安全的。
  state atomic.Uint64 // high 32 bits are value of counter, low 32 bits are waiter count.
  // WaitGroup 使用了信号量机制来实现 goroutine 的阻塞和唤醒。
  // 当等待的 goroutine 数量不为 0 时，调用 Wait 方法的 goroutine 会被阻塞，
  // 直到等待的 goroutine 数量变为 0 时，通过信号量唤醒所有等待的 goroutine。
  sema  uint32
}

// Note that it must not be embedded, due to the Lock and Unlock methods.
type noCopy struct{}

// Lock is a no-op used by -copylocks checker from `go vet`.
func (*noCopy) Lock()   {}
func (*noCopy) Unlock() {}

Add 方法会将传入的 delta 值累加到等待的 goroutine 数量上。如果 delta 为负数，相当于减少等待的 goroutine 数量。当累加后等待的 goroutine 数量变为 0 时，会唤醒所有正在等待的 goroutine。

计数器的值大于 0 或者 waiter 的数量等于 0， 不做处理直接返回。如果计数器值为 0 && 还有 waiter 被阻塞，则把 state 的计数清零，并唤醒那些被阻塞的 waiter。

func (wg *WaitGroup) Add(delta int) {
  if race.Enabled {
    if delta < 0 {
      // Synchronize decrements with Wait.
      race.ReleaseMerge(unsafe.Pointer(wg))
    }
    race.Disable()
    defer race.Enable()
  }
  state := wg.state.Add(uint64(delta) << 32)
  v := int32(state >> 32)
  w := uint32(state)
  if race.Enabled && delta > 0 && v == int32(delta) {
    // The first increment must be synchronized with Wait.
    // Need to model this as a read, because there can be
    // several concurrent wg.counter transitions from 0.
    race.Read(unsafe.Pointer(&wg.sema))
  }
  if v < 0 {
    panic("sync: negative WaitGroup counter")
  }
  if w != 0 && delta > 0 && v == int32(delta) {
    panic("sync: WaitGroup misuse: Add called concurrently with Wait")
  }
  if v > 0 || w == 0 {
    return
  }
  // This goroutine has set counter to 0 when waiters > 0.
  // Now there can't be concurrent mutations of state:
  // - Adds must not happen concurrently with Wait,
  // - Wait does not increment waiters if it sees counter == 0.
  // Still do a cheap sanity check to detect WaitGroup misuse.
  if wg.state.Load() != state {
    panic("sync: WaitGroup misuse: Add called concurrently with Wait")
  }
  // Reset waiters count to 0.
  wg.state.Store(0)
  for ; w != 0; w-- {
    runtime_Semrelease(&wg.sema, false, 0)
  }
}

Wait 方法用于阻塞当前 goroutine，直到所有等待的 goroutine 都完成工作。Wait 方法会检查等待的 goroutine 数量是否为 0。如果不为 0，当前 goroutine 会进入睡眠状态，直到等待的 goroutine 数量变为 0 时被唤醒。

检查计数器的值 v， v == 0 则返回，不发生阻塞，否则，利用原子操作 state, 把本 goroutine 加入 waiter 中，加入成功就阻塞等待被唤醒，反之，循环检查（可能有同时多个 waiter 调用 Wait 方法）。 

func (wg *WaitGroup) Wait() {
  if race.Enabled {
    race.Disable()
  }
  for {
    state := wg.state.Load()
    v := int32(state >> 32)
    w := uint32(state)
    if v == 0 {
      // Counter is 0, no need to wait.
      if race.Enabled {
        race.Enable()
        race.Acquire(unsafe.Pointer(wg))
      }
      return
    }
    // Increment waiters count.
    if wg.state.CompareAndSwap(state, state+1) {
      if race.Enabled && w == 0 {
        // Wait must be synchronized with the first Add.
        // Need to model this is as a write to race with the read in Add.
        // As a consequence, can do the write only for the first waiter,
        // otherwise concurrent Waits will race with each other.
        race.Write(unsafe.Pointer(&wg.sema))
      }
      runtime_Semacquire(&wg.sema)
      if wg.state.Load() != 0 {
        panic("sync: WaitGroup is reused before previous Wait has returned")
      }
      if race.Enabled {
        race.Enable()
        race.Acquire(unsafe.Pointer(wg))
      }
      return
    }
  }
}

Done 方法用于通知 WaitGroup 一个 goroutine 已经完成了它的工作。

// Done decrements the WaitGroup counter by one.
func (wg *WaitGroup) Done() {
  wg.Add(-1)
}

使用规则

1、确保 Add(delta) 和 defer Done() 的次数一致。

2、不能复制它。


errgroup.Group

与 Context 集成，error（如果有错误，返回第一个出现的错误，如果想要所有任务的错误，需要自行存储） 能向 Wait 传播。

使用

package main

import (
  "bufio"
  "fmt"
  "golang.org/x/sync/errgroup"
  "io"
  "io/fs"
  "os"
  "path/filepath"
)

type FileMergePipeline struct {
  FileReader
  FileWriter
}

type FileWriter struct {
  outFile *os.File
}

type FileReader struct {
  srcFileChan  chan string
  readGroup    *errgroup.Group
  maxChunkSize int
  numOfWorkers int
  transformer  chan []byte
}

type Option func(p *FileMergePipeline)

func WithMaxChunkSize(size int) Option {
  return func(p *FileMergePipeline) {
    p.FileReader.maxChunkSize = size
  }
}

func WithMaxNumOfWorker(n int) Option {
  return func(p *FileMergePipeline) {
    p.FileReader.numOfWorkers = n
  }
}

func WithOutFle(path string) Option {
  return func(p *FileMergePipeline) {
    outFile, err := os.Create(path)
    if err != nil {
      // ...
      return
    }
    p.FileWriter.outFile = outFile
  }
}

func NewFileReader(options ...Option) FileMergePipeline {
  p := FileMergePipeline{
    FileReader: FileReader{
      srcFileChan:  make(chan string),
      numOfWorkers: 10,
      maxChunkSize: 1024 * 20,
      readGroup:    &errgroup.Group{},
      transformer:  make(chan []byte),
    },
  }
  for _, opt := range options {
    opt(&p)
  }
  p.readGroup.SetLimit(p.numOfWorkers)
  return p
}

func (p *FileMergePipeline) Prepared() bool {
  return p != nil && p.outFile != nil
}

func (w *FileReader) Finish() {
  close(w.transformer)
}

func (w *FileReader) Read(path string) error {
  // 打开单个文件，读取
  file, fileErr := os.Open(path)
  if fileErr != nil {
    // ...
    return fileErr
  }
  defer func() {
    if err := file.Close(); err != nil {
      // ...
    }
  }()

  reader := bufio.NewReaderSize(file, w.maxChunkSize)
  for {
    buf := make([]byte, w.maxChunkSize)
    size, readErr := reader.Read(buf)
    if readErr != nil {
      // ...
      return readErr
    }
    if readErr == io.EOF {
      break
    }
    if size > 0 {
      w.transformer <- buf[:size]
    }
    if size < w.maxChunkSize {
      break
    }
  }
  return nil
}

func (p *FileMergePipeline) Write() error {
  // 阻塞等待，遍历 chan 不断写入目标文件
  for chunk := range p.FileReader.transformer {
    if _, err := p.FileWriter.outFile.Write(chunk); err != nil {
      // ...
      fmt.Println(`write err:`, err)
      return err
    }
  }

  if err := p.FileWriter.outFile.Close(); err != nil {
    fmt.Println(`dest file close err:`, err)
    return err
  }
  return nil
}

func main() {
  pipeline := NewFileReader(
    WithOutFle(`/Users/chenjianyang/Documents/Projects/Go/design_patterns/main/test1`),
    WithMaxNumOfWorker(1000),
    WithMaxChunkSize(1024*1024*3),
  )
  if !pipeline.Prepared() {
    // ...
    return
  }

  pipeline.readGroup.Go(func() error {
    defer close(pipeline.srcFileChan)
    return filepath.Walk(`/Users/chenjianyang/Documents/Projects/Go/design_patterns/main/testdata`, func(path string, info fs.FileInfo, err error) error {
      if info == nil || info.IsDir() {
        return nil
      }
      //
      if len(path) == 0 {
        return nil
      }
      fmt.Println(path)
      pipeline.srcFileChan <- path
      return nil
    })
  })
  // 并发 Map 阶段
  for path := range pipeline.srcFileChan {
    fmt.Println(`receive from srcChan: `, path)
    pipeline.readGroup.Go(func() error {
      defer func() {
        if r := recover(); r != nil {
          //
        }
      }()
      return pipeline.Read(path)
    })
  }

  // Reduce 阶段，阻塞等待有数据可以读入新的文件路径
  hereWeGo(func() {
    err := pipeline.Write()
    if err != nil {
      //...
    }
  })

  if err := pipeline.readGroup.Wait(); err != nil {
    // ...
  }
  pipeline.Finish() // 通知消费者，所有文件读取完毕
}

func hereWeGo(f func()) {
  go func() {
    defer func() {
      if r := recover(); r != nil {
        // print err
      }
    }()
    f()
  }()
}

使用规则：

1、对于并发任务，确定好 errorgroup 控制的对象。

2、如果并发数过多，使用 errorgroup.SetLimit() 控制。
