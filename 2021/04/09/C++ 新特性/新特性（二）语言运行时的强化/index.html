C++1x 新特性（二）语言运行时的强化

lambda

[捕获列表] (参数列表) mutable(可选) 异常属性 -> 返回类型 { // 函数体 }

#include <iostream>
#include <memory>

#define LOG(x) std::cout << x << std::endl

void TestReferenceCapture() {

  int value = 1;
  bool change = false;

  auto value_capture = [=] () {
      LOG("改变？" + change + " 值捕获： " + value);
  };
  auto reference_capture = [&] () {
      LOG("改变？" + change + " 引用捕获： " + value);
  };

  // 改变kvalue的值
  value = 10000;
  change = true;

  value_capture();
  reference_capture();
}

// C++14 表达式捕获
// 允许捕获的成员用任意的表达式进行初始化，允许右值捕获
void TestRvalueCapture() {
  std::unique_ptr<int> important = std::make_unique<int>(10000);

  // 泛型lambda since C++14, auto关键字产生意义上的泛型， 不需要 -> 返回类型
  auto add = [v1 = 1, v2 = std::move(important)] (int x, int y) /*-> int*/ {
      return (*v2) + (x + y + v1);
  };

  LOG(add(3, 4));
}

int main() {
  TestReferenceCapture();
  TestRvalueCapture();
  return 0;
}

move semantics（移动语义）

移动语义的出现解决了传统C++没有区分移动和拷贝而导致造成大量数据复制的问题。

#include <iostream>
#include <string>
#include <vector>

#define LOG(x) std::cout << x << std::endl

class A {
public:
 int* ptr_ = nullptr;

 A() : ptr_(new int(1)) {
       LOG("默认构造"  + ptr_ );
 }

 // 存在无意义的对象拷贝
 A(A& a) : ptr_(new int(*a.ptr_)) {
     LOG("拷贝构造"  + ptr_);
 }

 A(A&& a) : ptr_(a.ptr_) {
        a.ptr_ = nullptr;
        LOG("移动构造" + ptr_);
 }

 ~A() {
        LOG("析构" + ptr_);
        delete ptr_;
 }

};

// 这样定义能防止编译器优化
// 函数返回后会产生一个将亡值，被A的移动构造 A(A&&) 引用，从而延长生命周期，
// 并将这个右值中的指针拿到，保存到 A obj 中， 之后将亡值的指针会被置为 nullptr， 
// 防止了这块内存域被销毁，
// 从而别免了无意义的拷贝构造，增强性能。 
A Get_A(bool test) {
   A a, b;
   return test ? a : b;
}

int main() {

   A obj = Get_A(false);
   LOG(obj.ptr_);
   LOG(*obj.ptr_);

   // std::vector 例子
   std::string str = "xxx";
   std::vector<std::string> v;

   v.push_back(str);
   // str 为 "xxx"
   LOG("拷贝传参后的str: " + str);
   // 如果使用如下，str将被置为空。
   v.push_back(std::move(str));
   // str 为 ""
   LOG("移动传参后的str: " + str);

   return 0;
}

std::bind

#include <iostream>
#include <functional>

int Calculate(int x, int y, bool is_add) {
 	return is_add ? x + y : x * y;
}

// 将部分调用参数提前提前绑定到函数身上，成为一个新的对象，
// 然后在参数齐全后完成调用、
// 使用std::placeholders::_1对第3位参数进行占位，如果多个占位，依次为_1, _2, _3, ...

int main() {
     auto bind_calculate = std::bind(Calculate, 3, 4, std::placeholders::_1);
     std::cout << bind_calculate(false) << std::endl;
     return 0;
}


std::forward

完美转发, 为了在传递参数的时候保持原来的参数类型（左引用保持左引用，右引用保持右引用）


坍缩规则：

数形参类型 实参参数类型 推导后的函数形参类型
T& 左引用 T&
T& 右引用 T&
T&& 左引用 T&
T&& 右引用 T&& 

// 前面提到，声明一个右值引用其实是一个左值 std::string&& rs = "xx" + "yy";

#include <iostream>
#include <utility>

#define LOG(x) std::cout << x << std::endl

void log(int& v) {
	LOG("      左值引用");
}

void log(int&& v) {
	LOG("      右值引用");
}


template <typename T>
void pass(T&& v) {
    LOG("    --普通传参");
    log(v);
}

template <typename T>
void forward_pass(T&& v) {
    LOG("    --普通传参");
    log(v);

    // 单纯的将左值转化为右值。
    LOG("    --move 传参。");
    log(std::move(v));

    // 单纯将参数做了类型转换, 由std::forward的实现得知。
    /* 
  template<typename _Tp>
      constexpr _Tp&&
      forward(typename std::remove_reference<_Tp>::type& __t) noexcept
      { return static_cast<_Tp&&>(__t); }
  */
    LOG("    --forward 传参。");
    log(std::forward<T>(v));
}

int main() {
    LOG("传递右值:");
    pass(1); // 虽然传递右值1，但是对于pass(v)的参数v来说，v是一个（右值）引用，同时也是左值

    LOG("传递左值:");
    int value = 100;
    pass(value); // value是左引用，为什么能调用到pass(T&& v) ？ 如上表格坍缩规则。

    LOG("=================================");

    // 无论传递 左值（2）, 还是右值（f_value），普通传参都会将参数作为左值进行转发。
    // 所以std::move总会接到一个左值，从而转发调用 log(int&& )输出右值引用。
    // std::forward 没有造成多余的拷贝， 同时完美转发（传递）了函数实参给了内部调用的其他函数。
    LOG("传递右值:");
    forward_pass(2);

    LOG("传递左值:");
    int f_value = 200;
    forward_pass(f_value);

    return 0;
}

std::function

#include <iostream>
#include <functional>

#define LOG(x) std::cout << x << std::endl

// old
using foo = void(int);
// 函数指针的调用是非类型安全的
void excecute_f(foo f) {
    f(3);
}
void Oldman() {
    auto f = [](int value) {
        LOG(value);
    };
    excecute_f(f);
}


//new
int print(int para) {
 return para;
}
// 函数容器 将函数、函数指针作为对象处理，是一种通用、多态的函数封装，类型安全
void Coolman() {

    std::function<int(int)> func_1 = print;
    std::function<int(int)> func_2 = [&](int value) {
        return func_1(value) + value;
    };
    LOG(func_2(100));
}

int main() {
    Oldman();
    Coolman();
    return 0;
}
$ ./std_function 
3
200


std::move

左值: 符号左边的值，是表达式（不一定是赋值表达式）后依然存在的持久对象。

右值: 表达式结束后就不存在的临时对象。

纯右值: 纯粹的字面量，如 10, true, 1+2, ..

将亡值: 即将被销毁、却能够被移动的值。

#include <iostream>
#include <vector>
#include <string>
#include <utility>

#define LOG(x) std::cout << x << std::endl

std::vector<int> foo() {
    std::vector<int> tmp {1, 2, 3, 4};
    return tmp;
}

	// v是左值， foo()就是右值，也是纯右值
	// 对于foo()产生的返回值作为一个临时值，一旦被v复制后，就无法获取和修改
	std::vector<int> v = foo();

// std::move() 将左值参数无条件的转换成右值
void log(const std::string& str) {
	LOG("左值： " + str);
}

void log(const std::string&& str) {
	LOG("右值： " + str);
}

int main() {
    std::string ls1 = "xxx";
    // an rvalue reference cannot be bound to an lvalue
    // std::string&& rs2 = ls1;

    std::string&& rs2 = std::move(ls1);

    // 常量左值引用延长临时变量的生命周期， 但是ls3不可修改
    // initial value of reference to non-const must be an lvalue
    const std::string& ls3 = ls1 + ls1; 
    // ls3 += "yyy";

    // 右值引用延长生命周期，可以定义为非常量引用，可修改。
    std::string&& rs4 = ls1 + rs2;
    rs4 += "yyy";

    // rs4虽然引用了一个右值，但是由于它是一个引用，所以它依然是一个左值。
    log(rs4);
    return 0;
}
