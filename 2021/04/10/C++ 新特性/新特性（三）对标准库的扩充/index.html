std::array


std::array 和 std::vector 的区别

std::array std::vector
栈内存 堆内存
灵活访问，更高性能
更现代
编译时会创建一个固定的数组
不能够隐式转成指针（注意！！！）

​ 

#include <iostream>
#include <cstdio>
#include <array>
#include <algorithm>

// C风格的接口
void c_test (int* p, size_t len) {
 printf("%d, %d", *p, len);
}

int main() {
  std::array<int, 5> arr_1 = {1, 2, 3, 4, 5};

  // 无法转换成 C 风格指针传递！
  // no suitable conversion function from "std::array<int, 5UL>" to "int *" exists
  // c_test(arr_1, arr_1.size());

  // struct std::array<int, <error-constant>>
  // int len = 5;
  constexpr int len = 5;
  std::array<int, len> arr_2 = { 1, 3, 2, 5, 4 };
  std::sort(arr_2.begin(), arr_2.end());
  return 0;
}

std::tuple

#include <iostream>
#include <tuple>
#include <boost/variant.hpp>
#include <variant>

#define LOG(x) std::cout << x << std::endl

// 用于求取 tuple size

template <typename T>
auto tuple_size(const T& tpl) {
return std::tuple_size<T>::value;
}

// 借助 boost::variant 实现 
// 参考 https://www.boost.org/doc/libs/1_75_0/doc/html/variant.html
template <size_t n, typename... Args>
boost::variant<Args...> _tuple_index(size_t i, const std::tuple<Args...>& tpl) {
if (i == n) {
  return std::get<n>(tpl);
} else if (n == sizeof...(Args) - 1) {
  throw std::out_of_range("越界！");
}else {
  return _tuple_index<(n < sizeof...(Args) - 1 ? n + 1 : 0)>(i, tpl);
}
}

// 用于获取指定索引的 tuple 元素值
template <typename... Args>
boost::variant<Args...> tuple_index(size_t i, const std::tuple<Args...>& tpl) {
return _tuple_index<0>(i, tpl);
}

int main() {

auto student = std::make_tuple("Liming", 18, 'B');

constexpr int i_0 = 0, i_1 = 1, i_2 = 2;

// std::get<> 依赖的是编译期的常量
LOG("姓名： " + std::get<i_0>(student));
LOG("年龄： " + std::get<i_1>(student));
LOG("成绩： " + std::get<i_2>(student));

std::string name;
char grade;

// 利用std::tie 进行元组拆包， std::ignore忽略某个元素
std::tie(name, std::ignore, grade) = student;
LOG("姓名： " + name);
LOG("成绩： " + grade);

// 元组合并
auto teacher = std::make_tuple("Miss Li", 25, 'A');
auto concat = std::tuple_cat(std::move(student), std::move(teacher));

// 元组遍历
for (size_t i = 0; i != tuple_size(concat); ++i) {
  LOG(tuple_index(i, concat));
}

return 0;
}

std::unordered_map

#include <iostream>
#include <string>
#include <unordered_map>
#include <map>
#include <algorithm>

#define LOG(x) std::cout << x << std::endl

template <typename T>
void log(const T& t) {
LOG("容器类型： " + typeid(T).name());
std::for_each(t.begin(), t.end(), [](const auto& item){
  LOG("Key: " + item.first + "  Value: " + item.second);
});
}

int main() {
// 无序
std::unordered_map<int, std::string> unordered = {
  {1, "one"},
  {4, "four"},
  {3, "three"}, 
  {2, "two"}
};
// 有序
std::map<int, std::string> ordered = {
  {1, "one"},
  {4, "four"},
  {3, "three"}, 
  {2, "two"}
};

log(unordered);
log(ordered);

return 0;
}
