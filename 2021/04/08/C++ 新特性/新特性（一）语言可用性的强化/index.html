constexpr

(1) 拓宽「常量表达式」的范围。

(2)「显式『要求』表达式编译时求值」。告诉编译器 [我认为我是编译器常量]， 让编译器在需要编译器常量的语义处 [核实我说的是不是真的]。把它放到变量定义前，那么用来初始化这个变量的表达式「必须」是常量表达式，否则报错。

参考：https://www.zhihu.com/question/274323507

decltype

#include <iostream>

// 传统C++ 模板的错误写法
// template <typename R, typename T, typename U>
// R add(T t, U u) {
//     return t + u;
// }

// C++11 称之为尾返回类型（trailing return type）,利用 auto 关键字将返回值后置
template <typename T, typename U>
auto add (T t, U u) /*-> decltype(t + u) //C++14开始自动推导*/ {
	return t + u;
}


int main() {
 	auto result = add<float, double>(4.5, 5.6);

}

显示禁用默认函数

// 显示禁用默认函数
#include <iostream>

class Magic {
public:
   Magic() = delete; // 禁用编译器生成的构造
   Magic& operator=(const Magic&) = delete;
   Magic(const Magic& other) = delete;

   Magic(int a) {
         std::cout << "Not default constructor." << std::endl;
         e_ = a;
   }
   void tell() { std::cout << e_ << std::endl; }

private:
   int e_ = 0;
};

int main() {
 // the default constructor of "Inequality" 
 // cannot be referenced
 // -- it is a deleted function
 //   Inequality m1;
 //   m1.tell();

   Magic m2(2);
   m2.tell();

 // function "Magic::Magic(const Magic &other)" 
 // cannot be referenced
 //  -- it is a deleted function
 //   Magic m3 = m2;
 //   m3.tell();

 // function "Magic::operator=(const Magic &)" 
 // cannot be referenced 
 // -- it is a deleted function
 //   Magic m4(4);
 //   m4 = m2;
   return 0;
}

//继承构造
#include <iostream>

class Base {
public:
     Base() {
         v1_ = 1;
     }
     Base(int value) : Base() {
         v2_ = value;
     }

     virtual void tell() {
         std::cout << "v1: " << v1_ << "    v2: "<< v2_ << std::endl; 
     }
private:
     int v1_ = 0;
     int v2_ = 0;
};

// 构造函数如果需要继承是需要将参数一一传递的，这样会效率低下
// using 引入继承构造
class Derived : public Base {
public:
 	using Base::Base; // 继承构造
};

int main() {
    Derived derived(2222);
    derived.tell();
	return 0;
}

#include <iostream>

// 重载 operator<< 用来输出强类型枚举
template <typename T>
std::ostream& operator<<(typename std::enable_if<std::is_enum<T>::value, 
std::ostream>::type& stream, const T& e) {
 return stream << static_cast<typename std::underlying_type<T>::type>(e);
}

enum class Param {
     kValue1, 
     kValue2,
     kValue3 = 100,
     kValue4 = 100,
};

int main() {
    // no match for ‘operator==’ (operand types are ‘Param’ and ‘int’)

    if(Param::kValue3 == Param::kValue4) {

Variadic Template（可变长模板）

#include <iostream>
#include <string>

// 参数可以为0
template <typename... Ts>
class Magic {

};

// 参数至少为1
template <typename Required, typename... Args> 
class Magic_1 {

};

template <typename T>
void log(T info) {
 std::cout << info << std::endl;
}

template <typename... Args>
void printf(const std::string& str, Args... args);

template <typename... Args>
void magic(Args... args) {
  log(sizeof...(args));
}



template <typename T>
void printf(T value) {
  	log(value);
}

// 递归方式 解包
// 至少有2个参数时会调用此模板，当只有2个参数时，最终会调用printf<typename T>模板函数
template <typename T, typename... Args>
void printf(T value, Args... args) {
      std::cout << value << std::endl;
      printf(args...);
}

// 初始化列表展开 解包
// C++14 tricky
// 没有C++17之前都是这么干的：(void)std::initializer_list<int>{ (func(u), 0)... };
// 包展开语法：模式…，这里的模式是小括号中的部分。一个lambda表达式和value组成的逗号表达式

template <typename T>
void intializer_list_print(T value) {
 	std::cout << value << std::endl;
}

// (1)
template <typename T, typename... Args>
void intializer_list_print(T value, Args... args) {
     std::cout << value << std::endl;
     (void)std::initializer_list<int>{ (intializer_list_print(args), 0)... };
}

// (2) Brace-enclosed initializers
// 参考 https://en.cppreference.com/w/cpp/language/parameter_pack
template <typename T, typename... Args>
void intializer_list_print_2(T value, Args... args) {
 	std::cout << value << std::endl;

 	(void)std::initializer_list<T>{ (std::cout << args << std::endl, 0 )... };
}


// C++17 fold expression
// 参照libcxx的

// template <class ..._Tp>
// _LIBCPP_INLINE_VISIBILITY
// void __swallow(_Tp&&...) _NOEXCEPT {} 

// template <class _Tuple>
//         _LIBCPP_INLINE_VISIBILITY
//         typename enable_if
//         <
//             __tuple_assignable<_Tuple, tuple<_Tp...> >::value,
//             __tuple_impl&
//         >::type
//         operator=(_Tuple&& __t) _NOEXCEPT_((__all<is_nothrow_assignable<_Tp&, typename tuple_element<_Indx,
//                                        typename __make_tuple_types<_Tuple>::type>::type>::value...>::value))
//         {
//             __swallow(__tuple_leaf<_Indx, _Tp>::operator=(_VSTD::forward<typename tuple_element<_Indx,
//                                        typename __make_tuple_types<_Tuple>::type>::type>(_VSTD::get<_Indx>(__t)))...);
//             return *this;
//         }


// 所以可以类似：
// C++17 fold exprssion 参考 https://en.cppreference.com/w/cpp/language/fold
// __swallow(func(u)...);

template <typename T>
void cxx17_print(T value) {
 	std::cout << value << std::endl;
}

template <typename T, typename... Args>
void cxx17_print(T value, Args... args) {
 	std::cout << value << std::endl;
 	cxx17_print(args...);
}


int main() {
      log("递归方式：");
      printf(1, 2, "123", 1.1);

      log("初始化列表展开方式1：");
      intializer_list_print(1, 2, "123", 1.1f);

      log("初始化列表展开方式2：");
      intializer_list_print_2(1, 2, "123", 1.1f);

      log("C++17 fold expression:");
      cxx17_print(1, 2, "123", 1.1f);
      return 0;
}
